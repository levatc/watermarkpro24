  format?: 'original' | 'jpg' | 'png' | 'webp' | 'mp4';

  HIGH = 'high',

  success: boolean;

    jobId: string;
    status: JobStatus;
    estimatedDuration: number;
    queuePosition: number;
    files: JobFile[];

  };
}

enum JobStatus {
  PENDING = 'pending',
  PROCESSING = 'processing',
  COMPLETED = 'completed',
  FAILED = 'failed',
  CANCELLED = 'cancelled'
}

  fileId: string;
